<script>
(function(){
  var rIn=document.getElementById('rIn'), rOut=document.getElementById('rOut');
  var carIn=document.getElementById('carIn'), carInG=carIn.querySelector('g');
  var carOut=document.getElementById('carOut'), carOutG=carOut.querySelector('g');
  var DUR_IN=7000, PAUSE=1400, DUR_OUT=4500, GAP=900;
  var start=null;
  function place(path, car, carG, prog){
    var L=path.getTotalLength(), len=prog*L;
    var pt=path.getPointAtLength(len), ah=path.getPointAtLength(Math.min(len+1,L));
    var ang=Math.atan2(ah.y-pt.y, ah.x-pt.x)*180/Math.PI;
    car.setAttribute('transform','translate('+pt.x.toFixed(1)+' '+pt.y.toFixed(1)+')');
    carG.setAttribute('transform','rotate('+ang.toFixed(1)+')');
  }
  function ease(t){ return t<0.5 ? 2*t*t : 1-Math.pow(-2*t+2,2)/2; }
  place(rIn, carIn, carInG, 0); carOut.setAttribute('opacity','0'); place(rOut, carOut, carOutG, 0);
  var CYCLE=DUR_IN+PAUSE+DUR_OUT+GAP;
  function loop(ts){
    if(!start) start=ts;
    var e=(ts-start)%CYCLE;
    if(e<DUR_IN){ carIn.setAttribute('opacity','1'); place(rIn,carIn,carInG,ease(e/DUR_IN)); carOut.setAttribute('opacity','0'); }
    else if(e<DUR_IN+PAUSE){ carIn.setAttribute('opacity','1'); place(rIn,carIn,carInG,1); carOut.setAttribute('opacity','0'); }
    else if(e<DUR_IN+PAUSE+DUR_OUT){ carIn.setAttribute('opacity','0'); carOut.setAttribute('opacity','1'); place(rOut,carOut,carOutG,ease((e-DUR_IN-PAUSE)/DUR_OUT)); }
    else { carIn.setAttribute('opacity','0'); carOut.setAttribute('opacity','1'); place(rOut,carOut,carOutG,1); }
    requestAnimationFrame(loop);
  }
  requestAnimationFrame(loop);
})();
</script>
